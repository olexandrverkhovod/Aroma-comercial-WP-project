recent posts Used for the latest entries in the specified data type and category
 Parameters with values of duty:

'type' => 'post', - the type of data

'category' => '', - the categories of data in the specified data type

'num' => '5', - number of output records

'meta' => 'true', - whether or not the meta information (creation date, etc.)

'thumb' => 'true', - show the preview picture or not

'thumb_width' => '120' - the width of the image preview

'thumb_height' => '120' - the height of the image preview

'more_text_single' => '', - the text of the button to read more than

'excerpt_count' => '0', - number of words in a text to review the record

'custom_class' => '', - user's class for the main wrap posts

'custom_class_item' => '' - user's class wrapper for each post

[recent_posts type="post" category="" num="" meta="true" thumb="true" thumb_width="120" thumb_height="120" more_text_single="" excerpt_count="15" custom_class="" custom_class_item=""]
